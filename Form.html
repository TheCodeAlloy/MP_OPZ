// ============================================================ // KONFIGURACE — vyplňte před nasazením // ============================================================ const SHEET_ID = 'DOPLŇTE_ID_VAŠÍ_TABULKY'; } // ============================================================ // OTP — odeslání // ============================================================ function sendOTP(token) { if (!token || token.trim() === '') { return { success: false, error: 'Chybí přístupový token v odkazu.' }; } const employer = findEmployerByToken(token.trim()); if (!employer) return { success: false, error: 'Odkaz není platný nebo vypršel.' }; const email = employer.data[COL_EMAIL]; const name = employer.data[COL_NAME]; const otp = Math.floor(100000 + Math.random() * 900000).toString(); const expiry = new Date(); expiry.setMinutes(expiry.getMinutes() + OTP_VALIDITY_MIN); employer.sheet.getRange(employer.rowNum, COL_OTP + 1).setValue(otp); employer.sheet.getRange(employer.rowNum, COL_OTP_EXPIRY + 1).setValue(expiry.toISOString()); sendOTPEmail(email, name, otp); return { success: true, maskedEmail: maskEmail(email), employerName: name }; } // ============================================================ // OTP — ověření // ============================================================ function verifyOTP(token, otp) { if (!token || !otp) return { success: false, error: 'Chybí parametry.' }; const employer = findEmployerByToken(token.trim()); if (!employer) return { success: false, error: 'Neplatný token.' }; const storedOTP = employer.data[COL_OTP] ? employer.data[COL_OTP].toString().trim() : ''; const expiryString = employer.data[COL_OTP_EXPIRY] ? employer.data[COL_OTP_EXPIRY].toString() : ''; if (!storedOTP) return { success: false, error: 'Nejprve si nechejte zaslat přihlašovací kód.' }; const expiry = new Date(expiryString); if (isNaN(expiry.getTime()) || new Date() > expiry) { clearOTP(employer); return { success: false, error: 'Platnost kódu vypršela. Klikněte na „Zaslat nový kód".' }; } if (storedOTP !== otp.toString().trim()) { return { success: false, error: 'Nesprávný kód. Zkuste to znovu.' }; } clearOTP(employer); return { success: true, employerName: employer.data[COL_NAME], ico: employer.data[COL_ICO], accountNumber: employer.data[COL_ACCOUNT], email: employer.data[COL_EMAIL] }; } // ============================================================ // Odeslání formuláře — zápis do Odpovědi + 2 emaily // ============================================================ function submitForm(params) { // Bezpečnostní kontrola tokenu const employer = findEmployerByToken((params.token || '').trim()); const ss = SpreadsheetApp.openById(SHEET_ID); const answerSheet = ss.getSheetByName(ANSWERS_SHEET); if (!answerSheet) throw new Error('List "' + ANSWERS_SHEET + '" nebyl nalezen.'); answerSheet.appendRow(row); // Souhrn pro emaily const d = { employerName: params.employerName || '', ico: params.ico || '', accountNumber: params.accountNumber || '', cisloDohody: params.cisloDohody || '', zamestnanec: params.zamestnanec || '', idZamestnanec: params.idZamestnanec || '', mesic: params.mesic || '', rok: params.rok || '', hrubaMzda: Number(params.hrubaMzda) || 0, odvodPojistneho: Number(params.odvodPojistneho) || 0, nahradaPN: Number(params.nahradaPN) || 0, soucet: Number(params.soucet) || 0, datum: formatDate(now), email: params.email || '' }; // Email zaměstnavateli if (d.email) sendConfirmationEmail(d); // Email adminům const admins = getAdminEmails(); if (admins.length > 0) sendAdminNotification(d, admins); return { success: true }; } // ============================================================ // Email 1: Potvrzení zaměstnavateli // ============================================================ function sendConfirmationEmail(d) { const subject = 'Potvrzení přijetí mzdového příspěvku — ' + d.mesic + ' ' + d.rok; ${tableRow('Zaměstnavatel', d.employerName)} ${tableRow('IČO', d.ico)} ${tableRow('Bankovní účet', d.accountNumber)} ${tableRow('Číslo dohody', d.cisloDohody)} ${tableRow('Zaměstnanec', d.zamestnanec)} ${tableRow('ID v projektu', d.idZamestnanec)} ${tableRow('Období', d.mesic + ' ' + d.rok)} ${tableRow('Hrubá mzda', formatCZK(d.hrubaMzda))} ${tableRow('Odvod pojistného', formatCZK(d.odvodPojistneho))} ${tableRow('Náhrada mzdy za PN', formatCZK(d.nahradaPN))}

`; GmailApp.sendEmail(d.email, subject, '', { htmlBody: html, name: 'Mzdové příspěvky' }); } // ============================================================ // Email 2: Notifikace adminům // ============================================================ function sendAdminNotification(d, adminEmails) { const subject = '[Nový záznam] ' + d.employerName + ' — ' + d.mesic + ' ' + d.rok; const html = `

Administrátorská notifikace

`; adminEmails.forEach(email => { GmailApp.sendEmail(email.trim(), subject, '', { htmlBody: html, name: 'Systém mzdových příspěvků' }); }); } // ============================================================ // OTP email // ============================================================ function sendOTPEmail(to, name, otp) { const subject = 'Váš přihlašovací kód pro zadání mzdového příspěvku'; const html = `

Přihlašovací kód

Dobrý den,

obdrželi jsme žádost o přihlášení k formuláři mzdového příspěvku. Váš jednorázový kód je:

${otp}

⏱ Kód je platný ${OTP_VALIDITY_MIN} minut.

Pokud jste o přihlášení nežádali, tento email ignorujte.

Automatická zpráva — neodpovídejte na tento email.

`; GmailApp.sendEmail(to, subject, '', { htmlBody: html, name: 'Mzdové příspěvky' }); } // ============================================================ // Sdílené pomocné funkce // ============================================================ function findEmployerByToken(token) { const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = ss.getSheetByName(EMPLOYERS_SHEET); if (!sheet) throw new Error('List "' + EMPLOYERS_SHEET + '" nebyl nalezen.'); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][COL_TOKEN] && data[i][COL_TOKEN].toString().trim() === token) { return { rowNum: i + 1, data: data[i], sheet: sheet }; } } return null; } function clearOTP(employer) { employer.sheet.getRange(employer.rowNum, COL_OTP + 1).setValue(''); employer.sheet.getRange(employer.rowNum, COL_OTP_EXPIRY + 1).setValue(''); } function getAdminEmails() { const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = ss.getSheetByName(ADMINS_SHEET); if (!sheet) return []; return sheet.getDataRange().getValues() .map(r => r[0]).filter(e => e && e.toString().includes('@')); } function maskEmail(email) { const parts = email.split('@'); if (parts.length !== 2) return email; const name = parts[0]; const masked = name.length > 2 ? name.substring(0, 2) + '•'.repeat(Math.min(name.length - 2, 4)) : name[0] + '•'; return masked + '@' + parts[1]; } function formatCZK(amount) { return Number(amount).toLocaleString('cs-CZ') + ' Kč'; } function formatDate(date) { return Utilities.formatDate(date, Session.getScriptTimeZone(), 'dd.MM.yyyy HH:mm'); } function tableRow(label, value) { return '' + '' + label + '' + '' + (value || '—') + ''; } // ============================================================ // Testovací funkce — spusťte ručně v editoru // ============================================================ function TEST_sendOTP() { console.log(JSON.stringify(sendOTP('VÁŠ_TOKEN_ZDE'), null, 2)); } function TEST_verifyOTP() { console.log(JSON.stringify(verifyOTP('VÁŠ_TOKEN_ZDE', '123456'), null, 2)); }
